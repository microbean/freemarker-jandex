035/**
036 * A {@link BeanModel} that adapts an {@link IndexView} intelligently.
037 *
038 * @author <a href="https://about.me/lairdnelson"
039 * target="_parent">Laird Nelson</a>
040 *
041 * @see BeanModel
042 *
043 * @see IndexView
044 */
045public class IndexViewModel extends BeanModel {
046
047  private static volatile boolean beanInfoSearchPathSet;
048  
049  static {
050    setBeanInfoSearchPath();
051  }
052
053  protected final BeansWrapper beansWrapper;
054  
055  protected final IndexView indexView;
056  
057  public IndexViewModel(final BeansWrapper beansWrapper, final IndexView indexView) {
058    super(indexView, beansWrapper);
059    Objects.requireNonNull(beansWrapper);
060    Objects.requireNonNull(indexView);
061    this.beansWrapper = beansWrapper;
062    this.indexView = indexView;
063  }
064
065  @Override
066  public TemplateModel get(final String key) throws TemplateModelException {
067    TemplateModel returnValue = null;
068    if (key != null) {
069      final BeansWrapper beansWrapper = this.beansWrapper;
070      switch (key) {
071      case "allKnownImplementors":
072        returnValue = new AllKnownImplementorsModel(beansWrapper, this.indexView);
073        break;
074      case "allKnownSubclasses":
075        returnValue = new AllKnownSubclassesModel(beansWrapper, this.indexView);
076        break;
077      case "annotations":
078        returnValue = new AnnotationsModel(beansWrapper, this.indexView);
079        break;
080      case "classByName":
081        returnValue = new ClassByNameModel(beansWrapper, this.indexView);
082        break;
083      case "knownClasses":
084        returnValue = this.wrap(this.indexView.getKnownClasses());
085        break;
086      case "knownDirectImplementors":
087        returnValue = new KnownDirectImplementorsModel(beansWrapper, this.indexView);
088        break;
089      case "knownDirectSubclasses":
090        returnValue = new KnownDirectSubclassesModel(beansWrapper, this.indexView);
091        break;
092      default:
093        break;
094      }
095    }
096    if (returnValue == null) {
097      returnValue = super.get(key);
098    }
099    return returnValue;
100  }
101
102  private static final class AllKnownImplementorsModel extends IndexViewModel {
103
104    private AllKnownImplementorsModel(final BeansWrapper beansWrapper, final IndexView indexView) {
105      super(beansWrapper, indexView);
106    }
107
108    @Override
109    public final TemplateModel get(final String key) throws TemplateModelException {
110      TemplateModel returnValue = null;
111      if (key != null) {
112        returnValue = this.wrap(this.indexView.getAllKnownImplementors(DotName.createSimple(key)));
113      }
114      return returnValue;
115    }
116    
117  }

209  
210  static final void setBeanInfoSearchPath() {
211    synchronized (Introspector.class) {
212      if (!beanInfoSearchPathSet) {
213        final String[] beanInfoSearchPath = Introspector.getBeanInfoSearchPath();
214        final String[] newBeanInfoSearchPath;
215        if (beanInfoSearchPath == null || beanInfoSearchPath.length <= 0) {
216          newBeanInfoSearchPath = new String[1];
217        } else {
218          newBeanInfoSearchPath = Arrays.copyOf(beanInfoSearchPath, beanInfoSearchPath.length + 1);
219          assert newBeanInfoSearchPath != null;
220          assert newBeanInfoSearchPath.length > beanInfoSearchPath.length;
221        }
222        newBeanInfoSearchPath[newBeanInfoSearchPath.length - 1] = "org.microbean.freemarker.jandex.beaninfo";
223        Introspector.setBeanInfoSearchPath(newBeanInfoSearchPath);
224        beanInfoSearchPathSet = true;
225      }
226    }
227  }

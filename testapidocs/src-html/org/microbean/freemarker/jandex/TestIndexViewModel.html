017package org.microbean.freemarker.jandex;
018
019import java.beans.BeanInfo;
020import java.beans.FeatureDescriptor;
021import java.beans.IntrospectionException;
022import java.beans.Introspector;
023import java.beans.PropertyDescriptor;
024
025import java.util.Arrays;
026import java.util.Collections;
027import java.util.List;
028import java.util.Set;
029import java.util.TreeSet;
030
031import freemarker.template.Configuration;
032import freemarker.template.Version;
033
034import org.jboss.jandex.AnnotationInstance;
035import org.jboss.jandex.ClassInfo;
036import org.jboss.jandex.DotName;
037
038import org.junit.Before;
039import org.junit.Test;
040
041import static org.junit.Assert.assertEquals;
042import static org.junit.Assert.assertFalse;
043import static org.junit.Assert.assertNotNull;
044import static org.junit.Assert.assertTrue;
045
046public class TestIndexViewModel {
047
048  private static final Set<String> ANNOTATION_INSTANCE_PROPERTY_DESCRIPTOR_NAMES =
049    new TreeSet<>(Arrays.asList("class", "name", "target", "value", "values"));
050  
051  
052  private static final Set<String> CLASS_INFO_PROPERTY_DESCRIPTOR_NAMES =
053    new TreeSet<>(Arrays.asList("class", "classAnnotations", "classFacet",
054                                "enclosingClass", "enclosingMethod",
055                                "fieldFacet", "fields", "flags",
056                                "hasNoArgsConstructor", "interfaceNames",
057                                "interfaceTypes", "kind", "methodFacet",
058                                "methodParameterFacet",
059                                "methods", "name", "nestingType", "simpleName",
060                                "superClassType", "superName", "typeFacet",
061                                "typeParameters"));
062
063  private static final Set<String> DOT_NAME_PROPERTY_DESCRIPTOR_NAMES =
064    new TreeSet<>(Arrays.asList("class", "componentized", "local", "prefix"));
065  
066  /**
067   * A {@link Version} representing the version of <a
068   * href="http://freemarker.org/">Freemarker</a> in use.
069   *
070   * <p>This field is never {@code null}.</p>
071   */
072  static final Version FREEMARKER_VERSION = Configuration.VERSION_2_3_23;
073
074  private Configuration configuration;
075  
076  public TestIndexViewModel() {
077    super();
078  }
079
080  @Before
081  public void setUp() {
082    IndexViewModel.setBeanInfoSearchPath();
083    final Configuration configuration = new Configuration(FREEMARKER_VERSION);
084    configuration.setDefaultEncoding("UTF-8");
085    configuration.setAPIBuiltinEnabled(true);
086    this.configuration = configuration;
087  }
088
089  @Test
090  public void testBeanInfoStuff() {
091    IndexViewModel.setBeanInfoSearchPath();
092    assertBeanInfoSearchPathIsRight();
093  }
094
095  private static final void assertBeanInfoSearchPathIsRight() {
096    final String[] beanInfoSearchPath = Introspector.getBeanInfoSearchPath();
097    assertNotNull(beanInfoSearchPath);
098    final List<String> beanInfoSearchPathList = Arrays.asList(beanInfoSearchPath);
099    assertNotNull(beanInfoSearchPathList);
100    assertFalse(beanInfoSearchPathList.isEmpty());
101    assertEquals("org.microbean.freemarker.jandex.beaninfo", beanInfoSearchPathList.get(beanInfoSearchPathList.size() - 1));
102    assertEquals(beanInfoSearchPathList.indexOf("org.microbean.freemarker.jandex.beaninfo"),
103                 beanInfoSearchPathList.lastIndexOf("org.microbean.freemarker.jandex.beaninfo"));
104  }
105  
106  @Test
107  public void testThatBeanInfosArePickedUp() throws IntrospectionException {
108    assertBeanInfoSearchPathIsRight();
109    assertPropertyDescriptors(AnnotationInstance.class, ANNOTATION_INSTANCE_PROPERTY_DESCRIPTOR_NAMES);
110    // assertMethodDescriptors(AnnotationInstance.class, Collections.emptySet());
111    assertPropertyDescriptors(ClassInfo.class, CLASS_INFO_PROPERTY_DESCRIPTOR_NAMES);
112    assertPropertyDescriptors(DotName.class, DOT_NAME_PROPERTY_DESCRIPTOR_NAMES);    
113  }
114
115  private static final void assertMethodDescriptors(final Class<?> beanClass, final Set<? extends String> expectedNames) throws IntrospectionException {
116    assertNotNull(beanClass);
117    assertMethodDescriptors(Introspector.getBeanInfo(beanClass), expectedNames);
118  }
119  
120  private static final void assertMethodDescriptors(final BeanInfo beanInfo, final Set<? extends String> expectedNames) {
121    assertNotNull(beanInfo);
122    assertFeatureDescriptors(beanInfo.getMethodDescriptors(), expectedNames);
123  }
124  
125  private static final void assertPropertyDescriptors(final Class<?> beanClass, final Set<? extends String> expectedNames) throws IntrospectionException {
126    assertNotNull(beanClass);
127    assertPropertyDescriptors(Introspector.getBeanInfo(beanClass), expectedNames);
128  }
129  
130  private static final void assertPropertyDescriptors(final BeanInfo beanInfo, final Set<? extends String> expectedNames) {
131    assertNotNull(beanInfo);
132    assertFeatureDescriptors(beanInfo.getPropertyDescriptors(), expectedNames);
133  }
134  
135  private static final void assertFeatureDescriptors(final FeatureDescriptor[] fds, final Set<? extends String> expectedNames) {
136    assertNotNull(fds);
137    assertNotNull(expectedNames);
138    assertEquals(expectedNames.size(), fds.length);
139    for (final FeatureDescriptor fd : fds) {
140      assertNotNull(fd);
141      assertTrue(expectedNames.contains(fd.getName()));
142    }
143  }
144
145}
